	.data
prompt: .asciiz "Enter an integer: "
strans:	.asciiz "The answer is: "
newline:.asciiz "\n"

	.text
	.globl main
main:
	#initializations
	li	$s0,0
	li	$s1,0
	li	$s2,0
	li	$s4,0

	#for the loop counter
	li	$s3,4
	li	$s5,1

loop:
	#prompt the user for input
	li	$v0,4
	la	$a0,prompt
	syscall

	#read input
	li	$v0,5
	syscall
	move	$s0,$v0

        #check loop conditions                                                                                       
        beq     $s4,$s3,complete                                                                                     
                                                                                                                     
        #addition of inputs                                                                                          
        add     $s1,$s0,$s1                                                                                          
                                                                                                                     
        #loop counter                                                                                                
        add     $s4,$s5,$s4                                                                                          
                                                                                                                     
        #jump to beginning of loop                                                                                   
        j       loop                                                                                                 
                                                                                                                     
complete:                                                                                                            
                                                                                                                     
        change math operation                                                                                        
        mult    $s1,$s0                                                                                              
        mflo    $s2                                                                                                  
                                                                                                                     
        #the answer string                                                                                           
        li      $v0,4                                                                                                
        la      $a0,strans                                                                                           
        syscall                                                                                                      
                                                                                                                     
        #print the answer                                                                                            
        li      $v0,1                                                                                                
        move    $a0,$s2                                                                                              
        syscall                                                                                                      
                                                                                                                     
        #exit                                                                                                        
        li      $v0,10
